    def extract_legal_entities(self, citation_string):
        """
        Extracts Party A and Party B.
        Example: 'Smith v City of Johannesburg [2024] ZAGPJHC 12'
        Returns: ('Smith', 'City of Johannesburg')
        """
        # Regex to capture text between start and the first 'v' and between 'v' and citation markers
        match = re.search(r"([A-Z].+?)\s+v\s+([A-Z].+?)(?=\s*\[|\s*\d{4}|\s*\(|\s*SA|\s*All)", citation_string)
        if match:
            return match.group(1).strip(), match.group(2).strip()
        return None, None
